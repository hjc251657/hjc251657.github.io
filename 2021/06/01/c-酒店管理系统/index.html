bool deleteMember(T&t,Titi,int no)
{ if(findMember(t,ti,no)==false)
{ cout<<”\n\t该员工不存在，操作失败！”<<endl;
return false;
}
t.n–;
Tipi;
if(t.head->num==no)
{ pi=t.head->next;
delete(t.head);
t.head=pi;
return true;
}
for(ti=t.head;ti->next!=NULL&&((ti->next->num)!=no);ti=ti->next);
pi=ti->next->next;
delete(ti->next);
ti->next=pi;
return true;
}
